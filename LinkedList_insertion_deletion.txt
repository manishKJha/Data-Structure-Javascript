   constructor(data){
     this.data = data,
     this.next = null
   }
}

  constructor(){

  }
}

}

}

LinkedList.prototype.deleteNode = function(data){
  let temp = this.head;
  if(temp.next == null && temp.data == data){
     this.head = temp.next;
  }else{
    while(temp.next.data != data){
           temp = temp.next;
    }
    temp.next = temp.next.next;
  }
  return this;
}
